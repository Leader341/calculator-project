# calculator project 

# Inputs 

number = input("Enter two numbers and they will be added, subtracted and divided : ")

first_number = int(number [0])
second_number = int(number [1])

# Outputs 

print()

print(f"{first_number} + {second_number} = {first_number + second_number}")
print(f"{first_number} / {second_number} = {first_number / second_number}")
print(f"{first_number} - {second_number} = {first_number - second_number}")
